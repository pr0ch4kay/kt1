class User {
    private _name: string;
    private _login: string;
    private _password: string;
    private _grade: number;
    static count: number = 0;

    constructor(name: string, login: string, password: string, grade: number) {
        this._name = name;
        this._login = login;
        this._password = password;
        this._grade = grade;
        User.count++;
    }

    get name(): string {
        return this._name;
    }

        this._name = newName;
